Git Basics#

Git is a fast and modern implementation of version control

Common commands#

git init directory_name # create and turn the directory into a warehouse that git can manage

git add <file>.....
git add . # add all file to stage

git restore <file>... # rollback
git commit -m 'commit_info'
git status

git branch -d branch_name # delete
git branch # 查看当前所处分支
git branch branch_name # create new branch

git checkout branch_name # switch to brach-name
git checkout -b new_branch_name # create and switch to new branch

git merge branch_name # merge branch_name to current branch

git reset --hard HEAD^ # 
git reset --hard HEAD^^ # 
git resrt --hard HEAD~100 #
git reset --hard version_id

git log
git log -pretty=oneline

git stash

git diff <filename>

git remote # view the info of remote warehous
git remote -v # in detail

git remote add origin <http link>
git push -u origin master # the first time you connect the remotr warehouse and the local one
git push origin branch_name

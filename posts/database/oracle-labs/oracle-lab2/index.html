  1. 设计与建立上课考勤表Attend_d710，能登记每个学生的考勤记录包括正常、迟到、旷课、请假。能统计以专业为单位的出勤类别并进行打分评价排序，如迟到、旷课、请假分别扣2，5，1分。可以考虑给一初始的分值，以免负值。

每个学生的表，显然一条记录时指一个学生某一天的某一堂课的记录，用（学号+日期+课堂时间）来做主键，对应扣分放另外一张表，稍后对应着循环就好。

create table T_attend_d710
(
    sno varchar2(10),
    mno varchar2(2),
    sday date,
    unit char(2) check(unit in('12','34','56','78','90')),
    mstatus nchar(2) check(mstatus in('正常','请假','迟到','旷课')),
    constraint pk_T_attend primary key(sno,sday,unit),
    constraint fk_T_attend_sno foreign key(sno) references T_stud_d710(sno),
    constraint fk_T_attend_mno foreign key(mno) references T_major_d710(mno)
);

create table T_attendscore_d710
(
    mstatus nchar(2) primary key,
    score int
);

alter table T_stud_d710 add(sum_evaluation number);

  1. 建立个人考勤汇总表stud_attend与专业考勤表major_attend，表示每个学生或每个专业在某时间周期（起始日期，终止日期）正常、迟到、旷课、请假次数及考勤分值。
create table T_stud_attend_d710
(
    sno varchar2(10),
    stime date,
    etime date,
    normalcnt int,
    leavecnt int,
    latecnt int,
    absentcnt int,
    score number,
    constraint pk_T_stud_attend_time primary key(sno,stime,etime),
    constraint fk_T_stud_attend_sno foreign key(sno) references T_stud_d710(sno)
);
create table T_major_attend_d710
(
    mno varchar2(2),
    stime date,
    etime date,
    normalcnt int,
    leavecnt int,
    latecnt int,
    absentcnt int,
    score number,
    constraint pk_T_major_attend_time primary key(mno,stime,etime),
    constraint fk_T_major_attend_d710 foreign key(mno) references T_major_d710(mno)
);

5．建立触发器，当对考勤表Attend表进行相应插入、删除、修改时，对stud表的sum_evaluation 数值进行相应的数据更新。 Note:这里应该选用的是行级触发器，其会针对DML操作所影响的每一行数据都执行一次触发器。创建这种触发器时，必须在语法中使用FOR EACH ROW 插入对应加，删除对应减，修改对应有加有减

create trigger tg_T_attend_d710
    before insert or update or delete on T_attend_d710
    for each row
declare
    mnewscore binary_integer;
    moldscore binary_integer;
begin 
    if inserting then
        select score into mnewscore from T_attendscore_d710 where mstatus=:new.mstatus;
        update T_stud_d710 set sum_evaluation=nvl(sum_evaluation,0)+mnewscore where T_stud_d710.sno=:new.sno;
    elsif updating then
        select score into mnewscore from T_attendscore_d710 where mstatus=:new.mstatus;
        select score into moldscore from T_attendscore_d710 where mstatus=:old.mstatus;
        update T_stud_d710 set sum_evaluation=nvl(sum_evaluation,0)-moldscore where T_stud_d710.sno=:old.sno;
        update T_stud_d710 set sum_evaluation=nvl(sum_evaluation,0)+mnewscore where T_stud_d710.sno=:new.sno;
    else 
        select score into moldscore from T_attendscore_d710 where mstatus=:old.mstatus;
        update T_stud_d710 set sum_evaluation=nvl(sum_evaluation,0)-moldscore where T_stud_d710.sno=:old.sno;
    end if;
end;

6．建立过程，生成某专业某时段（起、止日期）的考勤汇总表major_attend中各字段值，并汇总相应专业，将考勤分值的汇总结果写入到major表中的sum_evaluation中。

create procedure P_attend_d710 (pmno char,pstime date,petime date)as
    pscore int:=0;
begin
    insert into T_major_attend_d710 values(pmno,pstime,petime,0,0,0,0,0);
    for cur in (
        select T_attend_d710.mstatus,count(T_attend_d710.mstatus)mstatus_cnt,sum(score)pscore
        from T_attend_d710,T_attendscore_d710
        where T_attend_d710.mstatus=T_attendscore_d710.mstatus and T_attend_d710.mno=pmno
        and sday>=pstime and sday<=petime
        group by T_attend_d710.mstatus) loop
        if cur.mstatus='正常' then
        update T_major_attend_d710 set normalcnt=cur.mstatus_cnt where mno=pmno and stime=pstime and etime=petime;
        elsif cur.mstatus='请假' then
        update T_major_attend_d710 set leavecnt=cur.mstatus_cnt where mno=pmno and stime=pstime and etime=petime;
        elsif cur.mstatus='迟到' then
        update T_major_attend_d710 set latecnt=cur.mstatus_cnt where mno=pmno and stime=pstime and etime=petime;
        elsif cur.mstatus='旷课' then
        update T_major_attend_d710 set absentcnt=cur.mstatus_cnt where mno=pmno and stime=pstime and etime=petime;
        end if;
        pscore:=pscore+cur.pscore;
    end loop;
    update T_major_attend_d710 set score=pscore where mno=pmno and stime=pstime and etime=petime;
    update T_major_d710 set sum_evaluation=pscore where mno=pmno;
end;

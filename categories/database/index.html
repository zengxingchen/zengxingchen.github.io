Lab 设计与建立上课考勤表Attend_d710，能登记每个学生的考勤记录包括正常、迟到、旷课、请假。能统计以专业为单位的出勤类别并进行打分评价排序，如迟到、旷课、请假分别扣2，5，1分。可以考虑给一初始的分值，以免负值。 每个学生的表，显然一条记录时指一个学生某一天的某一堂课的记录，用（学号+日期+课堂时间）来做主键，对应扣分放另外一张表，稍后对应着循环就好。 create table T_attend_d710 ( sno varchar2(10), mno varchar2(2), sday date, unit char(2) check(unit in('12','34','56','78','90')), mstatus nchar(2) check(mstatus in('正常','请假','迟到','旷课')), constraint pk_T_attend primary key(sno,sday,unit), constraint fk_T_attend_sno foreign key(sno) references T_stud_d710(sno), constraint fk_T_attend_mno foreign key(mno) references T_major_d710(mno) ); create table T_attendscore_d710 ( mstatus nchar(2) primary key, score int ); insert into T_attendscore_d710 values('正常',0); insert into T_attendscore_d710 values('请假',1); insert into T_attendscore_d710 values('迟到',2); insert into T_attendscore_d710 values('旷课',5); 为major表与stud表增加sum_evaluation 数值字段，以记录根据考勤表Attend_d710(Attendance)中出勤类别打分汇总的值。 alter table T_stud_d710 add(sum_evaluation number); alter table T_major_d710 add(sum_evaluation number); 建立个人考勤汇总表stud_attend与专业考勤表major_attend，表示每个学生或每个专业在某时间周期（起始日期，终止日期）正常、迟到、旷课、请假次数及考勤分值。 create table T_stud_attend_d710 ( sno varchar2(10), stime date, etime date, normalcnt int, leavecnt int, latecnt int, absentcnt int, score number, constraint pk_T_stud_attend_time primary key(sno,stime,etime), constraint fk_T_stud_attend_sno foreign key(sno) references T_stud_d710(sno) );create table T_major_attend_d710 ( mno varchar2(2), stime date, etime date, normalcnt int, leavecnt int, latecnt int, absentcnt int, score number, constraint pk_T_major_attend_time primary key(mno,stime,etime), constraint fk_T_major_attend_d710 foreign key(mno) references T_major_d710(mno) ); 4．根据major表中的值与stud中的值，为考勤表Attend输入足够的样本值，要求每个专业都要有学生，有部分学生至少要有一周的每天5个单元（12，34，56，78，90，没有课的单元可以没有考勤记录）的考勤完整记录，其中正常、迟到、旷课、请假 可以用数字或字母符号表示。...
